\newcommand{\gaussfunc}[2]{
	1/(sqrt(2*pi*(#1)^2))*exp(-(x-#2)^2/(2*#1^2))
}
\newcommand{\blackdottedgauss}{
	\addplot[
		domain=-5:5,
		samples=30,
		color=black,
		mark=*,
		only marks,
	]{
		\gaussfunc{1}{0}
	};
}

	\addplot[
		domain=-5:5,
		samples=60,
		color=green!60!black,

	};
}
